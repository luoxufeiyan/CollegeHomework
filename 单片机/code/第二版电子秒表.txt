#include<reg51.h>			
#define uchar unsigned char  
uchar table[]={0x3f,0x06,0x5b,0x4f,0x66,0x6d,0x7d,0x07,0x7f,0x6f};
uchar t;
uchar second;
void time0 (void) interrupt 1
{  
   
   TH0=0X3C;     //高8位
   TL0=0XB0;	 //低8位
   t++;
   if (t==20)
   {
   		t=0;  
		second++;
   }
   if (second==60) 
   		second=0;
   P0=table[second/10];
   P2=table[second%10];

   }
